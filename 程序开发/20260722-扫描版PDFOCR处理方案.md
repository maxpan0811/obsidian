# 扫描版 PDF OCR 处理方案

## 现有工具

- **qwen3-vl:30b** - 本地视觉模型，中文识别，无限制
- **百度 OCR API** - 云端 API，500次/天免费额度
- **PyMuPDF** - PDF 解析和图像渲染
- **DuckDB** - 结构化数据存储

## 处理流程

```

│ 扫描版 PDF  │
└──────┬──────┘
       │
       ▼
```

## 核心代码

### 1. 页面分类检测

```python
import fitz

def classify_pages(pdf_path):
    """检测每页是文本还是扫描"""
    doc = fitz.open(pdf_path)
    pages = []
    
    for page_num in range(len(doc)):
        page = doc[page_num]
        text = page.get_text().strip()
        
        if len(text) > 50:  # 有文字层
            pages.append({'type': 'text', 'page': page_num})
        else:  # 扫描页
            pages.append({'type': 'scan', 'page': page_num})
    
    doc.close()
    return pages
```

### 2. 扫描页 OCR (qwen3-vl)

```python
import fitz
import subprocess
import json

def ocr_with_qwen_vl(pdf_path, page_num):
    """用 qwen3-vl 识别扫描页"""
    # 渲染为图片
    doc = fitz.open(pdf_path)
    page = doc[page_num]
    pix = page.get_pixmap(matrix=fitz.Matrix(2, 2))  # 2x 缩放
    img_path = f'/tmp/page_{page_num}.png'
    pix.save(img_path)
    doc.close()
    
    # 调用 qwen3-vl
    result = subprocess.run(
        ['bash', '~/.claude/skills/qwen-vl-agent/scripts/qwen-vl.sh',
         img_path, '提取图片中的所有文字，保持格式'],
        capture_output=True, text=True, timeout=120
    )
    
    # 解析结果
    if '>>>RESULT_START' in result.stdout:
        text = result.stdout.split('>>>RESULT_START')[1].split('>>>RESULT_END')[0].strip()
        return {
            'page': page_num,
            'text': text,
            'source': 'qwen3-vl',
            'bbox': None  # 可后续补充
        }
    return None
```

### 3. 批量处理

```python
def process_scanned_pdfs(file_list):
    """批量处理扫描版 PDF"""
    results = []
    
    for pdf_path in file_list:
        print(f"处理: {pdf_path}")
        
        # 分类页面
        pages = classify_pages(pdf_path)
        scan_pages = [p for p in pages if p['type'] == 'scan']
        
        print(f"  扫描页: {len(scan_pages)}")
        
        # OCR 识别
        for page_info in scan_pages:
            result = ocr_with_qwen_vl(pdf_path, page_info['page'])
            if result:
                results.append({
                    'file': pdf_path,
                    **result
                })
    
    return results
```

### 4. 存入 DuckDB

```python
import duckdb

def save_to_duckdb(results):
    """存储 OCR 结果"""
    conn = duckdb.connect('~/Documents/artnova/duckdb/excel_data.duckdb')
    
    # 创建表
    conn.execute("""
        CREATE TABLE IF NOT EXISTS pdf_ocr_results (
            file_path VARCHAR,
            page_number INTEGER,
            text_content VARCHAR,
            ocr_source VARCHAR,
            processed_at TIMESTAMP
        )
    """)
    
    # 批量插入
    for r in results:
        conn.execute("""
            INSERT INTO pdf_ocr_results 
            VALUES (?, ?, ?, ?, CURRENT_TIMESTAMP)
        """, [r['file'], r['page'], r['text'], r['source']])
    
    conn.close()
```

## 查询示例

```sql
-- 搜索 OCR 内容
SELECT file_path, page_number, text_content
FROM pdf_ocr_results
WHERE text_content LIKE '%关键词%'
ORDER BY file_path, page_number;

-- 统计每个文件的 OCR 字数
SELECT file_path, SUM(LENGTH(text_content)) as total_chars
FROM pdf_ocr_results
GROUP BY file_path
ORDER BY total_chars DESC;
```
